{% set current_lang = multilingual::language_current() -%} {% set current_lang_is_default = multilingual::is_default_language() | int -%} {% set nav_langs = [] -%} {% set nav_links = [] -%} {# ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ -#} {# ## Process the "special" name '[langs]' -#} {# ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ -#} {% if "[langs]" == link_name and config.languages | length > 0 -%} {# ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ -#} {# ## {% endmacro element -%}
